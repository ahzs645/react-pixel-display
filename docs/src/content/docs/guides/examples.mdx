## Interactive scoreboard

The `text` prop is reactive — update it with state to build dynamic displays:

<LiveExample client:load code={`import { useState } from 'react';\nimport { PixelDisplay } from 'react-pixel-display';\n\nfunction Scoreboard() {\n  const [home, setHome] = useState(0);\n  const [away, setAway] = useState(0);\n  const text = \`\${String(home).padStart(2, '0')} - \${String(away).padStart(2, '0')}\`;\n\n  return (\n    <div>\n      <PixelDisplay text={text} foregroundColor="#ffcc00"\n        width={64} height={16} scale={6} glow />\n      <div style={{ display: 'flex', gap: 20, marginTop: 10 }}>\n        <div>\n          Home\n          <button onClick={() => setHome(h => Math.max(0, h - 1))}>−</button>\n          <button onClick={() => setHome(h => Math.min(99, h + 1))}>+</button>\n        </div>\n        <div>\n          Away\n          <button onClick={() => setAway(a => Math.max(0, a - 1))}>−</button>\n          <button onClick={() => setAway(a => Math.min(99, a + 1))}>+</button>\n        </div>\n        <button onClick={() => { setHome(0); setAway(0); }}>Reset</button>\n      </div>\n    </div>\n  );\n}`}>
  <ScoreboardDemo client:load />
</LiveExample>

## Effect switcher

Switch effects at runtime by changing the `effect` prop:

<LiveExample client:load code={`import { useState } from 'react';\nimport { PixelDisplay } from 'react-pixel-display';\nimport type { EffectName } from 'react-pixel-display';\n\nfunction EffectSwitcher() {\n  const [effect, setEffect] = useState<EffectName>('scroll_rtl');\n\n  return (\n    <div>\n      <PixelDisplay text="SWITCH EFFECTS" effect={effect} speed={40}\n        foregroundColor="#ff6600" width={96} height={16} scale={5} glow />\n      <button onClick={() => setEffect('scroll_rtl')}>Scroll</button>\n      <button onClick={() => setEffect('breeze')}>Breeze</button>\n      <button onClick={() => setEffect('rainbow_text')}>Rainbow</button>\n      <button onClick={() => setEffect('neon')}>Neon</button>\n    </div>\n  );\n}`}>
  <EffectSwitcherDemo client:load />
</LiveExample>

## Playback control

Use the ref API to build play/pause controls:

<LiveExample client:load code={`import { useRef, useState } from 'react';\nimport { PixelDisplay } from 'react-pixel-display';\nimport type { PixelDisplayRef } from 'react-pixel-display';\n\nfunction PlaybackControl() {\n  const ref = useRef<PixelDisplayRef>(null);\n  const [playing, setPlaying] = useState(true);\n\n  const toggle = () => {\n    if (ref.current?.isRunning()) {\n      ref.current.stop();\n      setPlaying(false);\n    } else {\n      ref.current?.start();\n      setPlaying(true);\n    }\n  };\n\n  return (\n    <div>\n      <PixelDisplay ref={ref} text="PLAY / PAUSE ME"\n        effect="scroll_rtl" speed={40} foregroundColor="#ff6600"\n        width={64} height={16} scale={6} glow />\n      <button onClick={toggle}>{playing ? 'Pause' : 'Play'}</button>\n    </div>\n  );\n}`}>
  <PlaybackDemo client:load />
</LiveExample>

## Custom pixel patterns

### Checkerboard

<LiveExample client:load code={`const size = 16;\nconst pixels: string[] = [];\nfor (let y = 0; y < size; y++) {\n  for (let x = 0; x < size; x++) {\n    pixels.push((x + y) % 2 === 0 ? '#ff6600' : '#000000');\n  }\n}\n\n<PixelDisplay pixels={pixels} width={size} height={size} scale={8} glow />`}>
  <CheckerboardDemo client:load />
</LiveExample>

### Color gradient

<LiveExample client:load code={`const width = 32, height = 16;\nconst pixels: string[] = [];\nfor (let y = 0; y < height; y++) {\n  for (let x = 0; x < width; x++) {\n    const t = x / (width - 1);\n    const r = Math.round(255 * (1 - t));\n    const b = Math.round(255 * t);\n    pixels.push(\`#\${r.toString(16).padStart(2, '0')}00\${b.toString(16).padStart(2, '0')}\`);\n  }\n}\n\n<PixelDisplay pixels={pixels} width={width} height={height} scale={6} glow />`}>
  <GradientDemo client:load />
</LiveExample>
