## Interactive scoreboard

The `text` prop is reactive — update it with state to build dynamic displays:

<ScoreboardDemo client:load />

```tsx
import { useState } from 'react';
import { PixelDisplay } from 'react-pixel-display';

function Scoreboard() {
  const [home, setHome] = useState(0);
  const [away, setAway] = useState(0);
  const text = `${String(home).padStart(2, '0')} - ${String(away).padStart(2, '0')}`;

  return (
    <div>
      <PixelDisplay text={text} foregroundColor="#ffcc00" width={64} height={16} scale={6} glow />
      <button onClick={() => setHome(h => h + 1)}>Home +1</button>
      <button onClick={() => setAway(a => a + 1)}>Away +1</button>
      <button onClick={() => { setHome(0); setAway(0); }}>Reset</button>
    </div>
  );
}
```

## Effect switcher

Switch effects at runtime by changing the `effect` prop:

<EffectSwitcherDemo client:load />

```tsx
import { useState } from 'react';
import { PixelDisplay } from 'react-pixel-display';
import type { EffectName } from 'react-pixel-display';

function EffectSwitcher() {
  const [effect, setEffect] = useState<EffectName>('scroll_rtl');

  return (
    <div>
      <PixelDisplay text="SWITCH EFFECTS" effect={effect} speed={40}
        foregroundColor="#ff6600" width={96} height={16} scale={5} glow />
      <button onClick={() => setEffect('scroll_rtl')}>Scroll</button>
      <button onClick={() => setEffect('breeze')}>Breeze</button>
      <button onClick={() => setEffect('rainbow_text')}>Rainbow</button>
      <button onClick={() => setEffect('neon')}>Neon</button>
    </div>
  );
}
```

## Playback control

Use the ref API to build play/pause controls:

<PlaybackDemo client:load />

```tsx
import { useRef, useState } from 'react';
import { PixelDisplay } from 'react-pixel-display';
import type { PixelDisplayRef } from 'react-pixel-display';

function PlaybackControl() {
  const ref = useRef<PixelDisplayRef>(null);
  const [playing, setPlaying] = useState(true);

  const toggle = () => {
    if (ref.current?.isRunning()) {
      ref.current.stop();
      setPlaying(false);
    } else {
      ref.current?.start();
      setPlaying(true);
    }
  };

  return (
    <div>
      <PixelDisplay ref={ref} text="PLAY / PAUSE ME"
        effect="scroll_rtl" speed={40} foregroundColor="#ff6600"
        width={64} height={16} scale={6} glow />
      <button onClick={toggle}>{playing ? 'Pause' : 'Play'}</button>
    </div>
  );
}
```

```tsx
const size = 16;

for (let y = 0; y < size; y++) {
  for (let x = 0; x < size; x++) {
```

### Color gradient

<GradientDemo client:load />

```tsx
const width = 32, height = 16;
const pixels: string[] = [];
for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    const t = x / (width - 1);
    const r = Math.round(255 * (1 - t));
    const b = Math.round(255 * t);
    pixels.push(`#${r.toString(16).padStart(2, '0')}00${b.toString(16).padStart(2, '0')}`);
  }
}

<PixelDisplay pixels={pixels} width={width} height={height} scale={6} glow />
```

```tsx
const width = 48, height = 12;
const pixels: string[] = [];

for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    const hue = (x / width) * 360;
    // Convert HSL to hex ...
    pixels.push(hslToHex(hue, 1, 0.5));
  }
}
```
